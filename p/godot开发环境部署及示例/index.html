void initialize_GodotTest_module(ModuleInitializationLevel p_level) {
    if (p_level == MODULE_INITIALIZATION_LEVEL_SCENE) {
        UtilityFunctions::print("Init :: Scene");
        GDREGISTER_CLASS(FowTexture);
        GDREGISTER_CLASS(Spline);
        GDREGISTER_CLASS(ProfilerContext);
        GDREGISTER_CLASS(ProfilerUI);




        GDREGISTER_CLASS(ProfilerTest);
    }

    if (p_level == MODULE_INITIALIZATION_LEVEL_EDITOR) {
        UtilityFunctions::print("Init :: Editor");
        GDREGISTER_CLASS(SplineGizmo);
        GDREGISTER_CLASS(SplineGizmoPlugin);
        GDREGISTER_CLASS(MyEditorPlugin);


        EditorPlugins::add_by_type<MyEditorPlugin>();

    }
    UtilityFunctions::print("Init :: None :: ", p_level);
}

        // return;

}

// Initialization.
GDExtensionBool GDE_EXPORT

                       GDExtensionInitialization *r_initialization) {
