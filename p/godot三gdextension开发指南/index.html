    UtilityFunctions::print(__FUNCTION__,"redraw::CCCCCCC :: ",get_node_3d()->get_name());
    clear();
    PackedVector3Array points;
    PackedInt32Array ids;
    PackedVector3Array lines;
    ERR_FAIL_COND(!SplineInstance);
    int numPoints = SplineInstance->GetControlPoints().size();
    ERR_FAIL_COND(numPoints == 0);

    for (int i = 0; i < numPoints; i++) {
        Vector3 CurPoint = SplineInstance->GetControlPoints()[i];
        points.append(CurPoint);
        ids.append(i);
    }

    int segNum = 100;
    Array TempP = SplineInstance->GetBezierCurve(segNum);
    for (int i = 0; i < segNum - 1; i++) {
        Vector3 CurPoint0 = TempP[i];
        Vector3 CurPoint1 = TempP[i + 1];
        lines.append(CurPoint0);
        lines.append(CurPoint1);

    add_handles(points, get_plugin()->get_material("HandleMat",this),ids);
    add_lines(lines, get_plugin()->get_material("RedMat",this));   
}
